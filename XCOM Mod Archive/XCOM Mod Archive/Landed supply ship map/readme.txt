Landed supply ship map for Long War by liquid

Adds a new map based on the crashed supply ship at the lake ("Overlook")

- extract the file "patch_clean_overlook.upk" in your install folder <...>/XCom-Enemy-Unknown/XEW/XComGame/CookedPCConsole
- create a backup of the DefaultMaps.ini found in <...>/XCom-Enemy-Unknown/XEW/XComGame/Config
- open DefaultMaps.ini and append the following 2 entries:

Maps=( MapFamily="CSupplyship_OverlookA",     Name="CSupplyship_OverlookA",        DisplayName="CSupplyship_OverlookA (Landed Transport) |5",  		MissionType=eMission_LandedUFO,         bInRotation=True,  TimeOfDay=eMissionTime_Night, ShipType=eShip_UFOSupply,     eRegion=eMissionRegion_None,         CouncilType=eFCMType_None,       NewMap=True, InitialChance=20,  InitialGain=1.04, DynamicAliens[0]=eChar_None,    DynamicAliens[1]=eChar_None, StreamingMaps[0]=( MapName="CIN_DropshipIntros", Loc=( X=0, Y=0, Z=0 ), Rot=( Pitch=0, Yaw=0, Roll=0 ) ), StreamingMaps[1]=( MapName="patch_ship_overlook", Loc=( X=0, Y=0, Z=0 ), Rot=( Pitch=0, Yaw=0, Roll=0 ) ), StreamingMaps[2]=( MapName="patch_clean_overlook", Loc=( X=0, Y=0, Z=0 ), Rot=( Pitch=0, Yaw=0, Roll=0 ) ), StreamingMaps[5]=( MapName="patch_hide_randomized_objects", Loc=( X=0, Y=0, Z=0 ), Rot=( Pitch=0, Yaw=0, Roll=0 ) ) )
